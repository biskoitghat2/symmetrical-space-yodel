## 3. ناوبری 4 جهته با Active Cell

### State Management:
```typescript
const [activeCell, setActiveCell] = useState<{row: number, col: number} | null>(null);

// Columns: 0=name, 1=qty, 2=price, 3=discount
const handleCellKeyDown = (e, row, col) => {
  switch(e.key) {
    case 'ArrowUp':
      setActiveCell({row: row - 1, col});
      break;
    case 'ArrowDown':
      setActiveCell({row: row + 1, col});
      break;
    case 'ArrowLeft':
      setActiveCell({row, col: col + 1}); // RTL
      break;
    case 'ArrowRight':
      setActiveCell({row, col: col - 1}); // RTL
      break;
    case 'Enter':
      // Move to next editable cell
      break;
  }
};
```

## 7. Refs برای Navigation

### ساختار:
```typescript
const cellRefs = useRef<{
  [rowIndex: number]: {
    [colIndex: number]: HTMLInputElement | null
  }
}>({});

// Usage:
cellRefs.current[rowIndex][colIndex]?.focus();
cellRefs.current[rowIndex][colIndex]?.select();
```

## 10. مثال کامل یک سلول

```tsx
<td 
  className={`
    p-0 border-l border-gray-200
    ${isActive ? 'bg-blue-50' : ''}
  `}
>
  <input
    ref={el => {
      if (!cellRefs.current[rowIndex]) {
        cellRefs.current[rowIndex] = {};
      }
      cellRefs.current[rowIndex][colIndex] = el;
    }}
    type="text"
    value={value}
    onChange={handleChange}
    onKeyDown={e => handleCellKeyDown(e, rowIndex, colIndex)}
    onFocus={e => {
      setActiveCell({row: rowIndex, col: colIndex});
      e.target.select();
    }}
    className="
      w-full h-full p-3
      text-base font-bold
      border-none outline-none
      focus:ring-2 focus:ring-blue-500
    "
  />
</td>
```
